import { DELETE } from "./route";

// mocks
jest.mock("@/lib/db", () => ({ dbConnect: jest.fn() }));
jest.mock("@/models/Room", () => ({ Room: { findByIdAndDelete: jest.fn() } }));

const { dbConnect } = require("@/lib/db");
const { Room } = require("@/models/Room");

beforeEach(() => {
  jest.resetAllMocks();
});

test("DELETE returns 200 and deletedId when room existed", async () => {
  const id = "room123";
  Room.findByIdAndDelete.mockResolvedValueOnce({ _id: id });

  const res: any = await DELETE(undefined as any, { params: { id } } as unknown);

  expect(dbConnect).toHaveBeenCalled();
  expect(Room.findByIdAndDelete).toHaveBeenCalledWith(id);

  expect(res.status).toBe(200);
  const body = await res.json();
  expect(body).toEqual({ ok: true, deletedId: id });
});

test("DELETE returns 404 when room not found", async () => {
  const id = "missing";
  Room.findByIdAndDelete.mockResolvedValueOnce(null);

  const res: any = await DELETE(undefined as any, { params: { id } } as unknown);

  expect(dbConnect).toHaveBeenCalled();
  expect(Room.findByIdAndDelete).toHaveBeenCalledWith(id);

  expect(res.status).toBe(404);
  const body = await res.json();
  expect(body).toEqual({ error: "Room not found" });
});

test("DELETE returns 500 on exception", async () => {
  const id = "err";
  Room.findByIdAndDelete.mockImplementationOnce(() => { throw new Error("boom"); });
  const errSpy = jest.spyOn(console, "error").mockImplementation(() => {});

  const res: any = await DELETE(undefined as any, { params: { id } } as unknown);

  expect(dbConnect).toHaveBeenCalled();
  expect(Room.findByIdAndDelete).toHaveBeenCalledWith(id);

  expect(res.status).toBe(500);
  const body = await res.json();
  expect(body).toEqual({ error: "Failed to delete room" });

  errSpy.mockRestore();
});
